# 图片懒加载
```html
<img v-img-lazy="item.picture">
```
``` javascript
import { useIntersectionObserver } from "@vueuse/core";
export const lazyPlug = {
  install(app) {
  //注册主键v-img-lazy
    app.directive("img-lazy", {
      // （触发元素，指令绑定值）
      mounted(el, binding) {
        // 监听元素是否进入视口  （监听元素  监听元素是否进入视口）
        useIntersectionObserver(el, (isIntersecting) => {
          if (isIntersecting) {
            el.src = binding.value;
            stop()
          }
        });
      },
    });
  },
};
```
``` javaScript
main.js 页面
import { lazyPlug } from './utils/Plug'
app.use(lazyPlug)
```

# 封装axios
目录建议结构
`src/
├── utils/
│   └── request.js          # 核心封装：创建实例 + 拦截器
├── api/
│   ├── modules/
│   │   ├── user.js         # 用户模块接口（使用 userApi 实例）
│   │   └── order.js        # 订单模块接口（使用 orderApi 实例）
│   └── index.js            # 统一导出所有实例和模块
└── main.js`
### 1.request.js文件
```js
import axios from 'axios'

const request=axios.create({

    baseURL:null,

    timeout:5000

})

// 请求拦截器

request.interceptors.request.use(

    config=>{

        return config
    },

    error=>{

        return Promise.reject(error)
    }

)

// 响应拦截器

request.interceptors.response.use(

    response=>{

        return response

    },

    error=>{

        return Promise.reject(error)

    }

)


export default request
```
